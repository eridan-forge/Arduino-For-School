# 🅿️ Урок 7: Парктроник — первый реальный прибор

**Тип урока:** теория + практика (финальный проект модуля)  
**Время:** 50–60 минут  
**Уровень:** продвинутый начинающий  
**Цель:** собрать полноценный парктроник — устройство, которое измеряет расстояние, показывает его на экране, меняет цвет индикации и подаёт звуковой сигнал.

Это **настоящий гаджет**, который можно поставить в машину или использовать как тренажёр для парковки.

### Парктроник умеет:
- 📏 измерять расстояние до препятствия
- 🖥️ показывать его на LCD экране
- 🔴🟢 менять цвет (зелёный → жёлтый → красный)
- 🔊 пищать, если близко
- 🔋 работать от повербанка, без компьютера

> **Это ваш первый реальный прибор. Гордитесь.**

## 🎯 Образовательная ценность

Проект объясняет:
| Предмет | Что показываем |
|---------|----------------|
| **Физика** | скорость звука 340 м/с, эхолокация, влияние температуры |
| **Математика** | формула `расстояние = (время × скорость) / 2` |
| **Программирование** | условия `if/else`, обработка сигналов, `pulseIn` |
| **Электроника** | работа датчиков, цветовая индикация, I2C шина |

---

## 🔌 Сборка (общая схема)

| Компонент       | Пин Arduino |
|----------------|-------------|
| HC‑SR04 Trig    | D9          |
| HC‑SR04 Echo    | D10         |
| LCD I2C         | SDA → A4, SCL → A5 |
| Красный LED     | D5 (резистор 220 Ом) |
| Зелёный LED     | D6 (резистор 220 Ом) |
| Пассивный зуммер| D3 (резистор 100 Ом) |

> ⚠️ **Важно:** здесь нет жёлтого светодиода. Мы используем два цвета:  
> - зелёный = безопасно  
> - красный = опасно  
> Жёлтый получается включением **обоих** диодов одновременно.  
> Звук подаётся только в красной зоне.

### Схема питания
- Все компоненты питаются от **5V** Arduino (или внешнего 5V через пин)
- При подключении к Power Bank парктроник работает автономно

## 💻 Полный код

```cpp
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// ========== НАСТРОЙКИ ==========
#define TRIG 9
#define ECHO 10
#define RED 5
#define GREEN 6
#define BUZZER 3

LiquidCrystal_I2C lcd(0x27, 16, 2);  // если не работает, поменяй на 0x3F

float distance;

// ========== SETUP ==========
void setup() {
  Serial.begin(9600);
  
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);
  pinMode(RED, OUTPUT);
  pinMode(GREEN, OUTPUT);
  pinMode(BUZZER, OUTPUT);
  
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("PARKTRONIC v2.0");
  delay(2000);
  lcd.clear();
  
  Serial.println("Парктроник запущен");
}

// ========== ИЗМЕРЕНИЕ РАССТОЯНИЯ ==========
float getDistance() {
  digitalWrite(TRIG, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  
  long duration = pulseIn(ECHO, HIGH, 30000);
  
  if (duration == 0) return 0.0;
  
  float cm = duration * 0.034 / 2;
  if (cm > 400) return 0.0;
  return cm;
}

// ========== ЗВУК И СВЕТ ==========
void updateIndicator(float dist) {
  // ЗОНА 1: ОПАСНО (0–20 см)
  if (dist > 0 && dist < 20) {
    digitalWrite(RED, HIGH);
    digitalWrite(GREEN, LOW);
    lcd.setCursor(0, 1);
    lcd.print("STOP!          ");
    tone(BUZZER, 1000);
    delay(100);
    noTone(BUZZER);
    delay(100);
  }
  // ЗОНА 2: ВНИМАНИЕ (20–50 см)
  else if (dist >= 20 && dist < 50) {
    digitalWrite(RED, HIGH);
    digitalWrite(GREEN, HIGH);   // красный + зелёный = жёлтый
    lcd.setCursor(0, 1);
    lcd.print("WARNING!       ");
    noTone(BUZZER);
    delay(200);
  }
  // ЗОНА 3: БЕЗОПАСНО (≥50 см)
  else if (dist >= 50) {
    digitalWrite(RED, LOW);
    digitalWrite(GREEN, HIGH);
    lcd.setCursor(0, 1);
    lcd.print("SAFE           ");
    noTone(BUZZER);
    delay(200);
  }
  // ЗОНА 4: НЕТ ОБЪЕКТА / ВНЕ ДИАПАЗОНА
  else {
    digitalWrite(RED, LOW);
    digitalWrite(GREEN, LOW);
    lcd.setCursor(0, 1);
    lcd.print("NO OBJECT      ");
    noTone(BUZZER);
    delay(200);
  }
}

// ========== ВЫВОД НА ЭКРАН ==========
void showDistance(float dist) {
  lcd.setCursor(0, 0);
  lcd.print("Dist: ");
  lcd.print(dist);
  lcd.print(" cm   ");
  
  // Образовательная строка (по желанию)
  // lcd.setCursor(0, 1);
  // lcd.print("V="); lcd.print(dist*0.034/2); lcd.print("ms");
}

// ========== LOOP ==========
void loop() {
  distance = getDistance();
  
  showDistance(distance);
  updateIndicator(distance);
  
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");
  
  delay(300);
}
